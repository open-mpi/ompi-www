327static __hwloc_inline int
328hwloc_plugin_check_namespace(const char *pluginname __hwloc_attribute_unused, const char *symbol __hwloc_attribute_unused)
329{
330#ifdef HWLOC_INSIDE_PLUGIN
331 void *sym;
332#ifdef HWLOC_HAVE_LTDL
333 lt_dlhandle handle = lt_dlopen(NULL);
334#elif defined(HWLOC_WIN_SYS)
335 HMODULE handle = GetModuleHandleA(NULL);
336#else
337 void *handle = dlopen(NULL, RTLD_NOW|RTLD_LOCAL);
338#endif
339 if (!handle)
340 /* cannot check, assume things will work */
341 return 0;
342#ifdef HWLOC_HAVE_LTDL
343 sym = lt_dlsym(handle, symbol);
344 lt_dlclose(handle);
345#elif defined(HWLOC_WIN_SYS)
346 sym = GetModuleHandleA("hwloc.dll");
347 FreeLibrary(handle);
348#else
349 sym = dlsym(handle, symbol);
350 dlclose(handle);
351#endif
352 if (!sym) {
353 static int verboseenv_checked = 0;
354 static int verboseenv_value = 0;
355 if (!verboseenv_checked) {
356 const char *verboseenv = getenv("HWLOC_PLUGINS_VERBOSE");
357 verboseenv_value = verboseenv ? atoi(verboseenv) : 0;
358 verboseenv_checked = 1;
359 }
360 if (verboseenv_value)
361 fprintf(stderr, "Plugin `%s' disabling itself because it cannot find the `%s' core symbol.\n",
362 pluginname, symbol);
363 return -1;
364 }
365#endif /* HWLOC_INSIDE_PLUGIN */
366 return 0;
367}

494static __hwloc_inline int
495hwloc_filter_check_pcidev_subtype_important(unsigned classid)
496{
497 unsigned baseclass = classid >> 8;
498 return (baseclass == 0x03 /* PCI_BASE_CLASS_DISPLAY */
499 || baseclass == 0x02 /* PCI_BASE_CLASS_NETWORK */
500 || baseclass == 0x01 /* PCI_BASE_CLASS_STORAGE */
501 || baseclass == 0x00 /* Unclassified, for Atos/Bull BXI */
502 || baseclass == 0x0b /* PCI_BASE_CLASS_PROCESSOR */
503 || classid == 0x0c04 /* PCI_CLASS_SERIAL_FIBER */
504 || classid == 0x0c06 /* PCI_CLASS_SERIAL_INFINIBAND */
505 || classid == 0x0502 /* PCI_CLASS_MEMORY_CXL */
506 || baseclass == 0x06 /* PCI_BASE_CLASS_BRIDGE with non-PCI downstream. the core will drop the useless ones later */
507 || baseclass == 0x12 /* Processing Accelerators */);
508}

526static __hwloc_inline int
527hwloc_filter_check_keep_object_type(hwloc_topology_t topology, hwloc_obj_type_t type)
528{
529 enum hwloc_type_filter_e filter = HWLOC_TYPE_FILTER_KEEP_NONE;
530 hwloc_topology_get_type_filter(topology, type, &filter);
531 assert(filter != HWLOC_TYPE_FILTER_KEEP_IMPORTANT); /* IMPORTANT only used for I/O */
532 return filter == HWLOC_TYPE_FILTER_KEEP_NONE ? 0 : 1;
533}
534
539static __hwloc_inline int
540hwloc_filter_check_keep_object(hwloc_topology_t topology, hwloc_obj_t obj)
541{
542 hwloc_obj_type_t type = obj->type;
543 enum hwloc_type_filter_e filter = HWLOC_TYPE_FILTER_KEEP_NONE;
544 hwloc_topology_get_type_filter(topology, type, &filter);
545 if (filter == HWLOC_TYPE_FILTER_KEEP_NONE)
546 return 0;
547 if (filter == HWLOC_TYPE_FILTER_KEEP_IMPORTANT) {
548 if (type == HWLOC_OBJ_PCI_DEVICE)
549 return hwloc_filter_check_pcidev_subtype_important(obj->attr->pcidev.class_id);
550 if (type == HWLOC_OBJ_OS_DEVICE)
551 return hwloc_filter_check_osdev_subtype_important(obj->attr->osdev.type);
552 }
553 return 1;
554}
